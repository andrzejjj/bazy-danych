EXEC dbo.sp_rename @objname = N'[dbo].[Posts]', @newname = N'Posts_old', @objtype = N'OBJECT'

SET ANSI_NULLS ON

CREATE TABLE [dbo].[Posts]
(
	[PST_ID] [int] NOT NULL,
	[PST_Content] [varchar](8000) COLLATE Latin1_General_100_CI_AS NOT NULL,
	[PST_CreationDate] [datetime] NOT NULL,
	[PST_USRID] [int] NOT NULL,
	[PST_THRID] [int] NOT NULL,

INDEX [IDX_PST_CREATIONDATE] NONCLUSTERED HASH 
(
	[PST_CreationDate]
)WITH ( BUCKET_COUNT = 16384),
INDEX [IDX_PST_THRID] NONCLUSTERED HASH 
(
	[PST_THRID]
)WITH ( BUCKET_COUNT = 1024),
INDEX [IDX_PST_USRID] NONCLUSTERED HASH 
(
	[PST_USRID]
)WITH ( BUCKET_COUNT = 1024),
CONSTRAINT [Posts_primaryKey] PRIMARY KEY NONCLUSTERED HASH 
(
	[PST_ID]
)WITH ( BUCKET_COUNT = 16384)
)WITH ( MEMORY_OPTIMIZED = ON , DURABILITY = SCHEMA_AND_DATA )
